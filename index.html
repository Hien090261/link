const express = require('express'); const cors = require('cors'); const { nanoid } = require('nanoid'); const fs = require('fs'); const path = require('path'); const app = express(); const PORT = process.env.PORT || 3000; const DB_FILE = path.join(__dirname, 'links.json'); // Middleware app.use(cors()); app.use(express.json()); app.use(express.static('public')); // Load/Init Database const loadLinks = () => { if (!fs.existsSync(DB_FILE)) { fs.writeFileSync(DB_FILE, JSON.stringify({})); return {}; } const data = fs.readFileSync(DB_FILE); return JSON.parse(data); }; const saveLink = (id, targetUrl) => { const links = loadLinks(); links[id] = targetUrl; fs.writeFileSync(DB_FILE, JSON.stringify(links, null, 2)); }; // Routes // 1. API Rút gọn link app.post('/api/shorten', (req, res) => { const { url } = req.body; if (!url) return res.status(400).json({ error: 'URL is required' }); // Simple URL validation try { new URL(url); } catch (e) { return res.status(400).json({ error: 'Invalid URL' }); } const id = nanoid(6); // Generate 6-char ID saveLink(id, url); res.json({ id, shortUrl: `${req.protocol}://${req.get('host')}/${id}` }); }); // 2. API Lấy link đích từ ID app.get('/api/target/:id', (req, res) => { const { id } = req.params; const links = loadLinks(); const target = links[id]; if (!target) return res.status(404).json({ error: 'Link not found' }); res.json({ target }); }); // 3. Route vượt link (trả về trang redirect.html) app.get('/:id', (req, res) => { const { id } = req.params; const links = loadLinks(); if (!links[id]) return res.status(404).send('Link not found'); // Phục vụ file redirect.html từ thư mục public res.sendFile(path.join(__dirname, 'public', 'redirect.html')); }); app.listen(PORT, () => { console.log(`Server is running at http://localhost:${PORT}`); });
